    public void addFood(String name) {
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues contentValues = new ContentValues();
        contentValues.put(food_name, name);
        contentValues.put(food_stock, 0);
        contentValues.put(food_price, 0);
        contentValues.put(food_chef, "");
        contentValues.put(food_sale, 0);
        db.insert(food_table, null, contentValues);
    }

    public void updateFood(String key, String name, int stock, int price, String chef, int sale) {
        SQLiteDatabase db = this.getWritableDatabase();
        ContentValues contentValues = new ContentValues();
        contentValues.put(food_name, name);
        contentValues.put(food_stock, stock);
        contentValues.put(food_price, price);
        contentValues.put(food_chef, chef);
        contentValues.put(food_sale, sale);
        db.update(food_table,contentValues,String.format("%s = ?", food_name), new String[] {key});
    }

    public void deleteFood(String key) {
        SQLiteDatabase db = this.getWritableDatabase();
        db.delete(food_table,String.format("%s = ?", food_name),new String[] {key});
    }

    public Cursor showFood() {
        SQLiteDatabase db = this.getWritableDatabase();
        String context = String.format("select * from %s;", food_table);
        return db.rawQuery(context, null);
        //return db.rawQuery("select * from " + food_table + ";", null);
    }
